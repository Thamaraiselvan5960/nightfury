Access Elements in HTML Document


	<!DOCTYPE html>
	<html>
	<head>
		<title>JS Tutorials</title>
	</head>
	<body>
		<p id="text">Hello World...!</p>

	<script type="text/javascript">
		//accessing an element
		var text_element = Document.getElementById('text'); 

		//styling the element
		text_element.style.background = 'blue';

		//adding a new attribute or update existing to the element
		//here attribute name is id
		text_element.setAttribute(id, text);

		//removing an attribute, here id is the attribute name
		text_element.removeAttribute(id);

	</script>
	</body>
	</html>
